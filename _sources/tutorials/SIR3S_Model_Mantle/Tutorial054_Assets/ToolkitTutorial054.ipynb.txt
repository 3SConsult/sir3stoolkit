   "cell_type": "code",
   "execution_count": 15,
   "id": "f7cae338",
   "metadata": {},
   "outputs": [],
   "source": [
    "# Create time vector: 0 to 600 seconds, step 20\n",
    "time_seconds = np.arange(0, 601, 20)\n",
    "\n",
    "# Create variable values between 0 and 100\n",
    "np.random.seed(42)\n",
    "werte = np.random.rand(len(time_seconds)) * 100\n",
    "\n",
    "YEAR = 2026\n",
    "DATE = datetime(YEAR, 1, 1)  # constant date for all entries\n",
    "\n",
    "rows = []\n",
    "\n",
    "for t, w in zip(time_seconds, werte):\n",
    "    timestamp = DATE + timedelta(seconds=int(t))\n",
    "    tag = timestamp.strftime(\"%Y-%m-%d\")\n",
    "    uhrzeit = timestamp.strftime(\"%H:%M:%S.%f\")\n",
    "\n",
    "    rows.append({\n",
    "        \"TAG\": tag,\n",
    "        \"UHRZEIT\": uhrzeit,\n",
    "        \"WERT\": w\n",
    "    })\n",
    "\n",
    "# Build DataFrame\n",
    "df = pd.DataFrame(rows, columns=[\"TAG\", \"UHRZEIT\", \"WERT\"])"
